##### 注册接口实现 (authController.js)

**代码清单 6-1**: 用户注册逻辑
```javascript
// back-end/routes/auth.js

const router = require('express').Router();
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const User = require('../models/User');

/**
 * POST /api/auth/register
 * 用户注册接口
 */
router.post('/register', async (req, res) => {
  try {
    const { email, password, username } = req.body;
    
    // 1.输入校验
    if (!email || !password || !username) {
      return res.status(400).json({
        success: false,
        message: '邮箱、密码和用户名不能为空'
      });
    }
    
    // 邮箱格式验证
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
      return res.status(400).json({
        success: false,
        message: '邮箱格式不正确'
      });
    }
    
    // 密码强度验证(至少8位,包含字母和数字)
    if (password.length < 8 || !/(?=.*[a-zA-Z])(?=.*\d)/.test(password)) {
      return res.status(400).json({
        success: false,
        message: '密码至少8位,且需包含字母和数字'
      });
    }
    
    // 2.检查邮箱是否已注册
    const existingUser = await User.findOne({ email });
    if (existingUser) {
      return res.status(409).json({
        success: false,
        message: '该邮箱已被注册'
      });
    }
    
    // 3.密码加密(saltRounds=10,约100ms计算时间)
    const saltRounds = 10;
    const hashedPassword = await bcrypt.hash(password, saltRounds);
    
    // 4.创建用户记录
    const newUser = new User({
      email,
      password: hashedPassword,
      username,
      isVerified: false,  // 需要邮箱验证(可选功能)
      createdAt: new Date()
    });
    
    await newUser.save();
    
    // 5.生成JWT Token并返回
    const token = jwt.sign(
      { userId: newUser._id, email: newUser.email },
      process.env.JWT_SECRET,
      { expiresIn: process.env.JWT_EXPIRE || '7d' }
    );
    
    res.status(201).json({
      success: true,
      message: '注册成功',
      data: {
        token,
        user: {
          id: newUser._id,
          email: newUser.email,
          username: newUser.username,
          avatar: newUser.avatar
        }
      }
    });
    
  } catch (error) {
    console.error('Register error:', error);
    res.status(500).json({
      success: false,
      message: '服务器内部错误,请稍后重试'
    });
  }
});

module.exports = router;
```

**代码要点说明**:
- **分层校验**: 前端格式验证 + 后端规则验证,双重保障
- **安全措施**: bcrypt加密存储,saltRounds=10平衡安全性与性能
- **错误处理**: 区分客户端错误(400/409)与服务端错误(500),返回友好提示
- **Token生成**: 注册成功即登录,返回JWT便于前后端状态同步

##### 登录接口实现

```javascript
/**
 * POST /api/auth/login
 * 用户登录接口
 */
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    // 1.查找用户
    const user = await User.findOne({ email }).select('+password');  // 显式包含password字段
    
    if (!user) {
      return res.status(401).json({
        success: false,
        message: '邮箱或密码错误'  // 不明确提示"邮箱不存在",防止枚举攻击
      });
    }
    
    // 2.验证密码
    const isMatch = await bcrypt.compare(password, user.password);
    
    if (!isMatch) {
      return res.status(401).json({
        success: false,
        message: '邮箱或密码错误'
      });
    }
    
    // 3.检查账号状态(可选:是否被封禁、是否验证邮箱等)
    if (!user.isVerified) {
      // 可选:强制要求先验证邮箱
      // return res.status(403).json({ success: false, message: '请先验证邮箱' });
    }
    
    // 4.签发JWT Token
    const token = jwt.sign(
      { 
        userId: user._id, 
        email: user.email 
      },
      process.env.JWT_SECRET,
      { expiresIn: process.env.JWT_EXPIRE || '7d' }
    );
    
    // 5.更新最后登录时间
    user.updatedAt = new Date();
    await user.save();
    
    res.json({
      success: true,
      message: '登录成功',
      data: {
        token,
        user: {
          id: user._id,
          email: user.email,
          username: user.username,
          avatar: user.avatar
        }
      }
    });
    
  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({
      success: false,
      message: '登录失败,请稍后重试'
    });
  }
});
```

**安全设计要点**:
- **防枚举攻击**: 统一返回"邮箱或密码错误",不区分"用户不存在"和"密码错误"
- **密码字段隔离**: User Schema中password设置select:false,查询时需显式.select('+password')
- **Token有效期**: 7天平衡安全性与便利性(生产环境可缩短至24h+Refresh Token机制)

#### 6.2.2 AI生成服务实现 (约600字)

##### 通义千问API封装 (qwenService.js)

**代码清单 6-3**: AI服务调用核心逻辑
```javascript
// back-end/services/qwenService.js

const https = require('https');

/**
 * 调用通义千问大模型API
 * @param {Object} options - 调用选项
 * @param {Array} options.messages - 消息数组(符合OpenAI格式)
 * @param {boolean} options.stream - 是否使用流式输出
 * @param {string} options.model - 模型名称(默认qwen-turbo)
 * @param {http.ServerResponse} options.res - HTTP响应对象(流式模式需要)
 * @returns {Promise<Object>} 非流式模式返回完整响应
 */
async function callQwenAPI(options) {
  const {
    messages,
    stream = true,
    model = process.env.QWEN_MODEL || 'qwen-turbo',
    res  // ServerResponse对象,用于SSE推送
  } = options;
  
  const apiKey = process.env.QWEN_API_KEY;
  const apiUrl = 'https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation';
  
  const requestBody = {
    model: model,
    input: {
      messages: messages
    },
    parameters: {
      result_format: 'message',  // 返回消息格式(非纯文本)
      temperature: 0.7,         // 创造性参数(0-1,越高越随机)
      top_p: 0.8,               // 核采样参数
      max_tokens: 2000,         // 最大生成长度(Design JSON通常<1500 tokens)
      enable_search: false       // 是否启用联网搜索(关闭以保证稳定性)
    }
  };
  
  if (stream) {
    // ========== 流式模式(SSE) ==========
    return new Promise((resolve, reject) => {
      const postData = JSON.stringify({ ...requestBody, parameters: { ...requestBody.parameters, incremental_output: true }});
      
      const urlObj = new URL(apiUrl);
      const reqOptions = {
        hostname: urlObj.hostname,
        port: 443,
        path: urlObj.pathname,
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${apiKey}`,
          'Content-Length': Buffer.byteLength(postData)
        }
      };
      
      const apiReq = https.request(reqOptions, (apiRes) => {
        let buffer = '';
        
        apiRes.on('data', (chunk) => {
          buffer += chunk.toString();
          
          // 解析SSE数据流
          const lines = buffer.split('\n');
          buffer = lines.pop();  // 保留未完成的行
          
          for (const line of lines) {
            if (line.startsWith('data:')) {
              const data = line.slice(5).trim();
              
              if (data === '[DONE]') {
                // 流结束
                res.write('data: [DONE]\n\n');
                res.end();
                resolve();
                return;
              }
              
              try {
                const parsed = JSON.parse(data);
                const content = parsed?.output?.choices?.[0]?.delta?.content;
                
                if (content) {
                  // 推送给前端(SSE格式)
                  const sseData = JSON.stringify({
                    type: 'content',
                    content: content
                  });
                  res.write(`data: ${sseData}\n\n`);
                }
                
                // 检查是否完成(finish_reason === 'stop')
                const finishReason = parsed?.output?.choices?.[0]?.finish_reason;
                if (finishReason === 'stop') {
                  // 尝试提取完整的Design JSON
                  const fullContent = parsed?.output?.choices?.[0]?.message?.content;
                  if (fullContent) {
                    const designJson = extractDesignJson(fullContent);
                    if (designJson) {
                      const sseFinal = JSON.stringify({
                        type: 'design_json',
                        designJson: designJson
                      });
                      res.write(`data: ${sseFinal}\n\n`);
                    }
                  }
                  
                  res.write('data: [DONE]\n\n');
                  res.end();
                  resolve();
                }
                
              } catch (e) {
                console.error('SSE parse error:', e);
              }
            }
          }
        });
        
        apiRes.on('end', () => {
          res.end();
          resolve();
        });
        
        apiRes.on('error', (err) => {
          console.error('API response error:', err);
          reject(err);
        });
      });
      
      apiReq.on('error', (err) => {
        console.error('API request error:', err);
        reject(err);
      });
      
      apiReq.setTimeout(30000, () => {
        apiReq.destroy();  // 30秒超时,终止请求
        reject(new Error('API request timeout'));
      });
      
      apiReq.write(postData);
      apiReq.end();
    });
    
  } else {
    // ========== 非流式模式(一次性返回) ==========
    // ... (类似实现,但等待完整响应后返回)
  }
}

/**
 * 从文本中提取JSON块(处理模型可能包裹在markdown代码块中的情况)
 */
function extractDesignJson(text) {
  // 策略1:尝试直接解析整个文本为JSON
  try {
    const json = JSON.parse(text);
    if (json.version && json.root) return json;  // 验证是Design JSON
  } catch (e) {}
  
  // 策略2:提取 ```json ... ``` 代码块
  const jsonBlockMatch = text.match(/```(?:json)?\s*([\s\S]*?)```/);
  if (jsonBlockMatch) {
    try {
      return JSON.parse(jsonBlockMatch[1]);
    } catch (e) {}
  }
  
  // 策略3:查找最外层的{...}括号对
  const braceMatch = text.match(/\{[\s\S]*\}/);
  if (braceMatch) {
    try {
      return JSON.parse(braceMatch[0]);
    } catch (e) {}
  }
  
  return null;  // 提取失败
}

module.exports = { callQwenAPI };
```

**代码要点说明**:
- **SSE协议实现**: 手动解析`data: {...}\n\n`格式,区分内容片段和最终结果
- **超时控制**: 30秒超时避免长时间挂起,释放资源
- **JSON提取策略**: 三层降级(直接解析→代码块→正则匹配),应对模型输出格式不稳定问题
- **错误处理**: 区分网络错误、解析错误、超时等不同情况

#### 6.3.1 DesignRenderer渲染器 (约500字)

**代码清单 6-5**: 递归渲染引擎完整实现
```jsx
// front-end/src/components/DesignRenderer/DesignRenderer.jsx

import React, { memo, useMemo } from 'react';
import './DesignRenderer.css';

/**
 * 组件类型到HTML标签的映射表
 */
const COMPONENT_TAG_MAP = {
  page: 'div',
  container: 'div',
  card: 'div',
  text: 'span',
  button: 'button',
  input: 'div',      // 包裹disabled的input元素
  image: 'div',      // 包裹img元素
  divider: 'hr',
  icon: 'span'
};

/**
 * DesignNode组件(React.memo优化)
 * 渲染单个设计节点及其子节点
 */
const DesignNode = memo(function DesignNode({ node, selectedId, onSelect }) {
  
  // 缓存样式转换结果,避免每次渲染重复计算
  const cssStyle = useMemo(() => {
    if (!node?.style) return {};
    
    const style = {};
    Object.entries(node.style).forEach(([key, value]) => {
      if (value === undefined || value === null) return;
      
      switch (key) {
        case 'padding':
        case 'margin':
          if (Array.isArray(value) && value.length === 4) {
            style[key] = `${value[0]}px ${value[1]}px ${value[2]}px ${value[3]}px`;
          }
          break;
          
        case 'width':
        case 'height':
        case 'gap':
        case 'fontSize':
        case 'borderRadius':
        case 'top':
        case 'left':
          style[key] = typeof value === 'number' ? `${value}px` : value;
          break;
          
        default:
          style[key] = value;
      }
    });
    
    return style;
  }, [node?.style]);
  
  // 判断是否被选中
  const isSelected = node?.id === selectedId;
  
  // 确定HTML标签
  const Tag = COMPONENT_TAG_MAP[node?.type] || 'div';
  
  // 组装className
  const classNames = [
    'design-node',
    `component-type-${node?.type}`,
    isSelected ? 'selected' : ''
  ].filter(Boolean).join(' ');
  
  // 处理特殊组件的内容渲染
  const renderContent = () => {
    switch (node?.type) {
      case 'text':
        return node.content || node.text || '';
        
      case 'button':
        return (
          <button 
            style={{ 
              width: '100%', 
              height: '100%',
              background: 'none',
              border: 'none',
              cursor: 'pointer',
              color: 'inherit',
              fontSize: 'inherit',
              fontWeight: 'inherit'
            }}
          >
            {node.content || node.text || '按钮'}
          </button>
        );
        
      case 'input':
        return (
          <input
            type="text"
            placeholder={node.placeholder || ''}
            disabled
            style={{
              width: '100%',
              height: '100%',
              border: 'none',
              outline: 'none',
              background: 'transparent',
              padding: '0 10px',
              fontSize: 'inherit',
              color: 'inherit'
            }}
          />
        );
        
      case 'image':
        return node.src ? (
          <img 
            src={node.src} 
            alt={node.alt || ''} 
            style={{ maxWidth: '100%', maxHeight: '100%', objectFit: 'cover' }}
            draggable={false}
          />
        ) : (
          <div className="image-placeholder">📷 图片</div>
        );
        
      default:
        return null;
    }
  };
  
  return (
    <Tag
      className={classNames}
      style={cssStyle}
      data-component-id={node?.id}
      data-component-type={node?.type}
      onClick={(e) => {
        e.stopPropagation();  // 阻止冒泡,避免父节点误触发
        onSelect?.(node);
      }}
    >
      {/* 递归渲染子节点 */}
      {node?.children?.map((child, index) => (
        <DesignNode
          key={child.id}
          node={child}
          selectedId={selectedId}
          onSelect={onSelect}
        />
      ))}
      
      {/* 渲染当前节点的内容 */}
      {renderContent()}
    </Tag>
  );
}, (prevProps, nextProps) => {
  // 自定义比较函数:仅当关键属性变化时才重渲染
  return prevProps.node?.id === nextProps.node?.id &&
         prevProps.selectedId !== nextProps.selectedId &&
         shallowEqual(prevProps.node?.style, nextProps.node?.style);
});

/**
 * 浅比较两个对象的一层属性
 */
function shallowEqual(obj1, obj2) {
  if (obj1 === obj2) return true;
  if (!obj1 || !obj2) return false;
  
  const keys1 = Object.keys(obj1);
  const keys2 = Object.keys(obj2);
  
  if (keys1.length !== keys2.length) return false;
  
  return keys1.every(key => obj1[key] === obj2[key]);
}

/**
 * DesignRenderer主组件
 * 接收完整的Design JSON并启动递归渲染
 */
function DesignRenderer({ designJson, onSelect, selectedId, isEmpty }) {
  
  if (!designJson?.root || isEmpty) {
    return (
      <div className="design-renderer-empty">
        <div className="empty-icon">🎨</div>
        <p>暂无设计稿</p>
        <p className="empty-hint">
          请通过文本描述或上传图片来生成设计稿
        </p>
      </div>
    );
  }
  
  return (
    <div className="design-renderer-canvas">
      <DesignNode
        node={designJson.root}
        selectedId={selectedId}
        onSelect={onSelect}
      />
    </div>
  );
}

export default DesignRenderer;
```

**代码要点说明**:
- **Memo优化**: 自定义比较函数,仅对比id、selectedId和style浅层,大幅减少不必要的重渲染
- **useMemo缓存**: 样式转换计算结果缓存,依赖数组[node.style]变化时才重新计算
- **事件冒泡控制**: stopPropagation()确保点击子节点不会触发父节点选中
- **组件映射表**: 清晰定义Design JSON类型到HTML标签的对应关系
- **空状态处理**: 无设计稿时显示友好的引导界面

#### 6.3.2 VisualEditor编辑器 (约400字)

**代码清单 6-6**: 编辑控制器核心逻辑
```jsx
// front-end/src/components/VisualEditor/VisualEditor.jsx

import React, { useState, useRef, useEffect, useCallback } from 'react';
import DesignRenderer from '../DesignRenderer/DesignRenderer';
import PropertyPanel from '../PropertyPanel/PropertyPanel';  // 属性面板(需自行实现)
import { findNodeById, updateNodeImmutable } from '../../utils/designJsonUtils';
import './VisualEditor.css';

/**
 * VisualEditor - 可视化编辑器主组件
 * 协调渲染器与属性面板,管理编辑状态
 */
function VisualEditor({ initialDesignJson, onChange, readOnly = false }) {
  
  // 使用固定Key确保组件实例稳定(解决选中状态丢失问题)
  const EDITOR_KEY = 'visual-editor-stable-key';
  
  // 本地状态:当前设计稿
  const [designJson, setDesignJson] = useState(initialDesignJson);
  
  // 使用Ref持久化选中状态(突变不触发重渲染)
  const selectedIdRef = useRef(null);
  
  // 同步外部传入的新设计稿(如从历史记录恢复)
  useEffect(() => {
    if (initialDesignJson && initialDesignJson !== designJson) {
      setDesignJson(initialDesignJson);
      
      // 验证选中节点是否仍存在于新树中
      const exists = findNodeById(initialDesignJson, selectedIdRef.current);
      if (!exists) {
        // 智能回退:选择第一个子节点或根节点
        selectedIdRef.current = initialDesignJson.root?.children?.[0]?.id || null;
      }
    }
  }, [initialDesignJson]);
  
  /**
   * 处理组件选中
   */
  const handleSelect = useCallback((node) => {
    if (!node) return;
    
    // 更新Ref(不触发重渲染)
    selectedIdRef.current = node.id;
    
    // 强制刷新一次以更新属性面板(通过计数器触发)
    setSelectionCounter(prev => prev + 1);  // 需要额外state驱动属性面板更新
  }, []);
  
  /**
   * 处理属性更新(局部更新,避免全树替换)
   */
  const handleUpdateNode = useCallback((nodeId, updates) => {
    setDesignJson(prevJson => {
      // 使用Immutable方式更新,仅修改目标节点路径上的对象
      const newJson = updateNodeImmutable(prevJson, nodeId, updates);
      
      // 通知父组件(如全局Store)
      onChange?.(newJson);
      
      return newJson;
    });
  }, [onChange]);
  
  /**
   * 处理删除组件
   */
  const handleDeleteNode = useCallback((nodeId) => {
    setDesignJson(prevJson => {
      const newJson = deleteNodeImmutable(prevJson, nodeId);
      onChange?.(newJson);
      
      // 清除选中状态(如果删除的是当前选中节点)
      if (selectedIdRef.current === nodeId) {
        selectedIdRef.current = null;
      }
      
      return newJson;
    });
  }, [onChange]);
  
  // 获取当前选中的节点数据
  const selectedNode = selectedIdRef.current 
    ? findNodeById(designJson, selectedIdRef.current)?.node 
    : null;
  
  return (
    <div className="visual-editor" key={EDITOR_KEY}>
      {/* 设计画布区域 */}
      <div className="editor-canvas">
        <DesignRenderer
          designJson={designJson}
          selectedId={selectedIdRef.current}
          onSelect={handleSelect}
          isEmpty={!designJson}
        />
      </div>
      
      {/* 属性面板(右侧或底部浮动) */}
      {!readOnly && selectedNode && (
        <PropertyPanel
          key={selectedNode.id}  // 选中节点变化时重新挂载
          selectedNode={selectedNode}
          onUpdate={(updates) => handleUpdateNode(selectedNode.id, updates)}
          onDelete={() => handleDeleteNode(selectedNode.id)}
        />
      )}
    </div>
  );
}

export default VisualEditor;
```

**关键技术点**:
- **稳定Key策略**: EDITOR_KEY使用固定字符串,不再基于JSON内容生成,避免组件重建
- **Ref持久化选中状态**: selectedIdRef的修改不触发React重渲染,但在读取时能获取最新值
- **Effect同步机制**: 外部designJson变化时,自动同步本地状态并验证选中节点有效性
- **不可变更新**: updateNodeImmutable采用深拷贝+路径更新,保证数据不可变性
